import plotly.graph_objects as go
import plotly.express as px
import plotly.io as pio

pio.renderers.default = "plotly_mimetype+notebook_connected"
# INITIALIZE GRAPH OBJECT
fig = go.Figure()


# Set the continent values
cont = ['Oceania',
        'North America',
        'Europe',
        'South America',
        'Asia',
        'Africa']

# Set the color scheme
scheme = ['rgb(166, 216, 84)',
          'rgb(231, 138, 195)',
          'rgb(141, 160, 203)',
          'rgb(255, 217, 47)',
          'rgb(252, 141, 98)',
          'rgb(102, 194, 165)']

# add trace for every situation
j = 0
for i in cont:
    df4 = df.query("continent==@i")
    fig.add_trace(  # Add A trace to the figure
    go.Box(  # Specify the type of the trace
        x=df4["year"],  # Data-x
        y=df4["Happiness Score"],  # Data-y
        name = i,
        boxmean=True,
        marker_color = scheme[j],
        opacity=1.0,
        visible=True,  # Specify whether or not to make data-visible when rendered
    )
    )
    j+=1



fig.update_layout(
    xaxis = dict(
        tickvals = list(set(df['year']))
    ),
    plot_bgcolor = 'rgba(0,0,0,0)',
    title = dict(
    text = 'Happiness Score from 2008 to 2021 Among Different Continents',
    x = 0.05,
    y = 0.85
    ),
    xaxis_title = 'Time (Year)',
    yaxis_title = 'Happiness Score',

    boxmode = 'group',
    
    legend_title_text = "Continent"
)
fig.update_yaxes(
                 showline=True, linewidth=1, gridcolor='lightgrey',
                 layer='below traces')

fig.show()
fig.write_html("../img/Boxplot_Happiness.html")
