# bubble plot with x is log gpt per capita and y is happiness score. It has a time slider from 2008-2021. Group by continent and the size of the bubble is pupulation size.
fig = px.scatter(merged_df, x="Log GDP per capita", y="Happiness Score", animation_frame="year", animation_group="Country",
           size="Population", color="continent", hover_name="Country", size_max=70, color_discrete_map={
               "Oceania": "rgb(166, 216, 84)", #set color
               "North America": "rgb(231, 138, 195)",
               "Europe": "rgb(141, 160, 203)",
               "South America": "rgb(255, 217, 47)",
               "Asia": "rgb(252, 141, 98)",
               "Africa": "rgb(102, 194, 165)",
           }, category_orders={ #set order
               "continent": ["Oceania", "North America", "Europe", "South America", "Asia", "Africa"]
           })

fig.update_layout(title="GDP Per Capita vs Happiness Score ",
                  xaxis_title="Log GDP Per Capita (Dollar)",
                  yaxis_title="Happiness Score (range 0 to 10)",
                  template="plotly_white",
